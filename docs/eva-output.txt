function M.ask(messages, options)
    local opts = options or {}

    local request_body = {
        model = opts.model or config.model,
        messages = messages,
        temperature = opts.temperature or config.default_temperature,
        max_tokens = opts.max_tokens or config.default_max_tokens,
        stream = false
    }

    local request_json = json.encode(request_body)

    local response_body = {}
    local res, code, headers = http.request{
        url = config.base_url .. "/chat/completions",
        method = "POST",
        headers = {
            ["Content-Type"] = "application/json",
            ["Authorization"] = "Bearer " .. config.api_key,
            ["Content-Length"] = tostring(#request_json)
        },
        source = ltn12.source.string(request_json),
        sink = ltn12.sink.table(response_body)
    }

    if code ~= 200 then
        return nil, "HTTP " .. tostring(code) .. ": " .. table.concat(response_body, "")
    end

    local response_text = table.concat(response_body, "")
    local response_data = json.decode(response_text)

    if not response_data or not response_data.choices or #response_data.choices == 0 then
        return nil, "Invalid response from DeepSeek API"
    end

    return response_data.choices[1].message.content
end

        if provider_config.base_url and type(provider_config.base_url) ~= "string" then

-- LLM Provider Facade
-- Routes requests to configured provider

local log = require("core.log")

local M = {}

-- Provider cache
local current_provider = nil
local provider_name = nil

-- Load provider module
local function load_provider(name)
    local ok, provider = pcall(require, "providers." .. name)
    if not ok then
        log.error("Failed to load provider '" .. name .. "': " .. tostring(provider))
        return nil, "Provider '" .. name .. "' not found: " .. tostring(provider)
    end

    if type(provider) ~= "table" or type(provider.ask) ~= "function" then
        return nil, "Provider '" .. name .. "' does not implement required interface"
    end

    return provider
end

-- Get configured provider name
local function get_provider_name()
    -- 1. Check environment variable
    local env_provider = os.getenv("EVA_LLM_PROVIDER")
    if env_provider and env_provider ~= "" then
        return env_provider:lower()
    end

    -- 2. Default to deepseek
    return "deepseek"
end

-- Initialize provider
local function init_provider()
    local name = get_provider_name()

    if current_provider and provider_name == name then
        return true -- Already initialized
    end

    local provider, err = load_provider(name)
    if not provider then
        log.error("Failed to load LLM provider: " .. tostring(err))
        return nil, err
    end

    -- Load provider-specific config
    local config = {}
    -- TODO: Load from config file if exists

    local ok, init_err = provider.init(config)
    if not ok then
        log.error("Failed to initialize provider '" .. name .. "': " .. tostring(init_err))
        return nil, init_err
    end

    current_provider = provider
    provider_name = name

    log.info("LLM provider initialized: " .. name)
    return true
end

-- Main ask function (preserves existing API)
function M.ask(messages, options)
    -- Initialize provider if needed
    local ok, err = init_provider()
    if not ok then
        return nil, "LLM provider initialization failed: " .. tostring(err)
    end

    -- Validate messages
    if not messages or type(messages) ~= "table" or #messages == 0 then
        return nil, "No messages provided"
    end

    -- Call provider's ask function
    return current_provider.ask(messages, options)
end

-- Utility function to get current provider info
function M.get_provider_info()
    if not current_provider then
        return nil, "No provider initialized"
    end

    return {
        name = provider_name,
        has_init = type(current_provider.init) == "function",
        has_validate = type(current_provider.validate_config) == "function"
    }
end

    if not response then
        return nil, "Test failed: " .. tostring(err)
    end

    return "Test passed: " .. response:sub(1, 100)
end

return M
